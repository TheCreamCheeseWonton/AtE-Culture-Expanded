
###Americanist add Detroit 

americanist_add_detroit_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	}
	decision_group_type = major

	ai_check_interval = 120

	desc = americanist_add_detroit_decision_desc
	selection_tooltip = americanist_add_detroit_decision_tooltip

	is_shown = {
		# Standard filter checks.
		is_landed = yes
		# Must be americanist faith, and decision has not been taken yet.
		faith = {
			religion = religion:americanist_religion
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:americanist_detroit_added
			}
		}
	}

	is_valid = {
		#independent
		is_independent_ruler = yes
		#Own Detroit
		has_title = title:c_detroit
		#have decent enough piety/prestige levels
		prestige_level >= 3
		piety_level >= 3
		#Detroit must be your faith (lakeshore)
		title:c_detroit = {
			faith = ROOT.faith
		}
	}
		

	is_valid_showing_failures_only = {
		is_available_at_peace_adult = yes
	}

	effect = {
		#save scope
		#save_scope_as = detroit_lakeshore_restorer
		# Send an event.
		#trigger_event = ate_religious_decision.lakeshore.detroit
		#activate the site
		faith:libertarian = {
			activate_holy_site = americanist_detroit
		}
		faith:constitutionalist = {
			activate_holy_site = americanist_detroit
		}
		faith:exceptionalist = {
			activate_holy_site = americanist_detroit
		}
		faith:heartful = {
			activate_holy_site = americanist_detroit
		}
		faith:imaginerian = {
			activate_holy_site = americanist_detroit
		}
		faith:mayflower = {
			activate_holy_site = americanist_detroit
		}
		faith:principlist = {
			activate_holy_site = americanist_detroit
		}
		faith:yeomanist = {
			activate_holy_site = americanist_detroit
		}
		#give the player some renown
		dynasty = {
			add_dynasty_prestige = medium_dynasty_prestige_value
		}
		#and a nickname
		give_nickname = nick_the_lightbringer
		#add the flag to prevent it from being repeated
		hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:americanist_detroit_added
			}
		}
	}

	cost = { 
		piety = 750
		prestige = 750
		gold = 500
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 20
	}
}

	}

	parameters = {
		architect_trait_prestige = yes

		cultural_head_fascination_mult = 0.1

	}

	cost = {

			add = {
				value = tradition_base_cost

				limit = {
					NOT = { has_innovation = innovation_development_02 }
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = lacks_development_02_innovation_desc
				}
			}

			multiply = tradition_replacement_cost_if_relevant
		}
	}

	ai_will_do = {
		value = 50
		add = {
			value = scope:character.ai_sociability

		}
	}

		0 = steward
		1 = mediterranean
		4 = sericulture.dds
	}

	is_shown = {
		# DLC check.
		has_ep3_dlc_trigger = yes
		OR = {
			AND = {
				exists = culture_head
				this = culture_head
				culture_head.house = { has_house_modifier = sericulture_domicile_house_modifier }	
			}
			any_culture_county = {
				title_province = {
					OR = {
						geographical_region = world_brazil_sudeste

				text = culture_parent_silk_tradition

				format = "BASE_VALUE_FORMAT"
			}

					NOT = {
						any_culture_county = {
							percent >= 0.4
							any_county_province = {

									geographical_region = ghw_region_parana

							}
						}
					}

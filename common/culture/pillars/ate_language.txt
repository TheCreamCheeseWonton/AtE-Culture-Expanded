language_abenaki = {
    type = language
    is_shown = {
        language_is_shown_trigger = {
            LANGUAGE = language_abenaki
        }
    }
    parameters = {
        language_group_eastern_algonquian = yes
        language_family_algic = yes 
    }
    ai_will_do = {
        value = 10
        if = {
            limit = { has_cultural_pillar = language_abenaki }
            multiply = same_language_choice_factor
        }
        else_if = {
            limit = { has_cultural_parameter = language_group_eastern_algonquian }
            multiply = same_language_group_choice_factor
        }
        else_if = {
            limit = { has_cultural_parameter = language_family_algic }
            multiply = same_language_family_choice_factor
        }
    }
    color = {  114 142 224  }
}

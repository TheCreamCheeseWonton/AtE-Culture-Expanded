## Americanist Presidency
can_run_for_americanist_presidency_trigger = {
	# Basic Triggers
	age >= 35
	faith = { has_doctrine_parameter = follows_americanist_presidency }
	NAND = { exists = var:refuse_to_be_election_candidate has_variable = refuse_to_be_election_candidate }
	NOT = { has_title = title:k_presidency }
	is_alive = yes

	OR = { # Either you're a count+, a baron/mayor that's famous/pious, or a courtier that's REALLY famous or pious, or a patrician
		AND = {
			is_landed = yes
			OR = {
				is_independent_ruler = yes # prevent border gore (a better system will be developed later)
				highest_held_title_tier = tier_barony
			}
			OR = {
				highest_held_title_tier >= tier_county
				prestige_level >= 3
				piety_level >= 3
			}
		}
		prestige_level >= 5
		piety_level >= 5
		# is_patrician = yes # This trigger doesn't exist
	}
}

can_form_republic_trigger = {
	OR = {
		#either your faith allows it...
		AND = {
			faith = { has_doctrine_parameter = can_form_republic }
			#...you are very pious...
			piety_level >= 4
			NOT = {
				#...and none of your powerful vassals are either...
				any_powerful_vassal = {
					OR = {
						#...people who disapprove you or...
						opinion = {
							target = root
							value < 0
						}
						#...members of a nonrepublican faith...
						faith = { NOT = { has_doctrine_parameter = can_form_republic } }
					}
				}
			}

		}
		#...or your culture allows it...
		AND = {
			culture = { has_cultural_parameter = can_form_republic }
			#...and you are very prestigious...
			prestige_level >= 4
			NOT = {
				#...and none of your powerful vassals...
				any_powerful_vassal = {
					OR = {
						#...either dislikes you or...
						opinion = {
							target = root
							value < 0
						}
						#...is from a culture that doesn't support republics...
						culture = { NOT = { has_cultural_parameter = can_form_republic } }
					}
				}
			}
		}
		#TODO add here titles that should allow it
		#...or you are...
		AND = {
			#...someone who admires presidentialism...
			OR = {
				has_trait = scholar
				has_trait = diplomat
			}
			#...and is not afraid of trying weird stuff...
			has_trait = eccentric
			#...and very prestigious...
			prestige_level >= 4
			#...and has maximum authority
			has_realm_law_flag = max_authority_level
		}
		
		#or you are commonfolk!
		AND = {
			OR = {
				has_trait = populist_leader
				has_trait = peasant_leader
			}
			NOT = {
				#...and none of your powerful vassals...
				any_powerful_vassal = {
				
					opinion = {
						target = root
						value < 0
					}
					#only check for unhappy nobles
					#if you're forming it as a folk leader,
					#other folk leaders should support you
					is_lowborn = no
					has_trait = populist_leader
					has_trait = peasant_leader
				}
			}
		}
	}
}

can_form_administrative_trigger = {
	has_ep3_dlc_trigger = yes
	OR = {
		AND = {
			highest_held_title_tier = tier_kingdom
			culture = { has_cultural_parameter = can_adopt_administrative_government_as_kingdom_tier }
		}
		highest_held_title_tier = tier_empire
	}
	trigger_if = {
		limit = {
			is_independent_ruler = yes
			is_ai = no
		}
		prestige_level >= 3
	}
	trigger_if = {
		limit = {
			is_independent_ruler = yes
			is_ai = no
		}
		custom_tooltip = {
			text = all_powerful_vassals_required_tt
			any_powerful_vassal = {
				count = all
				opinion = {
					target = root
					value >= 50
				}
			}
		}
	}
	trigger_if = {
		limit = { is_ai = yes }
		calc_true_if = {
			amount >= 2
			culture = { has_cultural_pillar = ethos_bureaucratic }
			culture = { has_cultural_pillar = ethos_courtly }
			culture = { has_cultural_pillar = ethos_egalitarian }
			culture = { has_cultural_tradition = tradition_legalistic }
			culture = { has_cultural_tradition = tradition_hereditary_hierarchy }
			has_realm_law = crown_authority_3
			capital_county = {
				development_level > medium_development_level
			}
			faith = { has_doctrine = tenet_legalism }
			dynasty = {
				has_dynasty_perk = erudition_legacy_5
			}
			dynasty = {
				has_dynasty_perk = law_legacy_5
			}
		}
	}
}

favorite_treat_location_trigger = {
	trigger_if = {
		limit = {
			exists = capital_province
			is_travelling = no
		}
		capital_province = {
			geographical_region = $REGION$
		}
	}
	trigger_else_if = {
		limit = {
			exists = liege.capital_province
			is_travelling = no
		}
		liege.capital_province = {
			geographical_region = $REGION$
		}
	}
	trigger_else_if = {
		limit = {
			exists = location
		}
		location = {
			geographical_region = $REGION$
		}
	}
	trigger_else = {
		always = no
	}
}

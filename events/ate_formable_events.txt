#I formed Hispaniola!
ate_form.0001 = { 
	type = character_event
	title = form_hispaniola.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					OR = {
						root.culture = culture:haitian
					}
				}
				desc = form_hispaniola.0001.desc.haitian
			}
			triggered_desc = {
				trigger = {
					OR = {
						root.culture = culture:quisqueyano
					}
				}
				desc = form_hispaniola.0001.desc.quisqueyano
			}
		}
		desc = form_hispaniola.0001.start.desc
	}
	
	theme = realm
	left_portrait = {
		character = scope:hispaniola_former
		animation = admiration
	}

	immediate = {
		play_music_cue = "mx_cue_positive_effect"

	}

	option = {
		name = form_hispaniola.0001.a
		hidden_effect = {
			if = {
				limit = {
					scope:hispaniola_former.faith = {
						OR = {
							AND = {
								religion_tag = catholic_religion
								NOT = { has_doctrine = tenet_afro_diasporic_syncretism }
							}
							AND = {
								religion_tag = afro_diasporic_religion
								OR = {
									NOT = { has_doctrine = tenet_catholic_syncretism }
									NOT = { has_doctrine = tenet_christian_syncretism }
								}
							}
						}
					}
				}
				trigger_event = ate_form.1001
			}
		}
	}
}

#Reunite Connecticut

ate_form.0033 = {
	type = character_event
	title = reunite_connecticut.t
	desc = reunite_connecticut.desc
	theme = realm
	left_portrait = {
		character = scope:connecticut_unifier
		animation = personality_honorable
	}
	immediate = {
		play_music_cue = "mx_cue_epic_sacral_moment"
		title:d_connecticut.de_jure_liege = {
			save_scope_as = connecticut_de_jure
		}
		title:d_new_haven.de_jure_liege = {
			save_scope_as = new_haven_de_jure
		}
	}
	option = { # 
		name = reunite_connecticut.connecticut
		trigger = {
			NAND = {
				NOT = { scope:connecticut_de_jure = scope:new_haven_de_jure }
				OR = {
					is_vassal_of = scope:new_haven_de_jure.holder
					is_vassal_of = scope:new_haven_de_jure.de_jure_liege.holder
				}
			}
		}
		title:d_connecticut = {
			set_de_jure_liege_title = scope:connecticut_de_jure
		}
		title:d_new_haven = {
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_county }
				set_de_jure_liege_title = title:d_connecticut
			}
		}
		destroy_title = title:d_new_haven
	}
	option = {
		name = reunite_connecticut.new_haven
		trigger = {
			#always = yes
			NOT = {
				scope:connecticut_de_jure = scope:new_haven_de_jure
			}
			OR = {
				is_independent_ruler = yes
				is_vassal_of = scope:new_haven_de_jure.holder
				is_vassal_of = scope:new_haven_de_jure.de_jure_liege.holder
			}
		}
		title:d_connecticut = {
			set_de_jure_liege_title = scope:new_haven_de_jure
		}
		title:d_new_haven = {
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_county }
				set_de_jure_liege_title = title:d_connecticut
			}
		}
		destroy_title = title:d_new_haven
	}

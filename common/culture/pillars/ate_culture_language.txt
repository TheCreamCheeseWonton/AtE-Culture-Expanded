## ANYTIME YOU ADD A LANGUAGE GROUP, FAMILY, OR UNION:
## YOU MUST ADD THAT GROUP/FAMILY/UNION TO ccu_scripted_effects.txt

language_ancient_english = {
    type = language
    is_shown = {
        language_is_shown_trigger = {
            LANGUAGE = language_ancient_english
        }
    }
    parameters = {
        language_group_ancient_english = yes
        language_family_englic = yes 
    }
    ai_will_do = {
        value = 10
        if = {
            limit = { has_cultural_pillar = language_ancient_english }
            multiply = same_language_choice_factor
        }
        else_if = {
            limit = { has_cultural_parameter = language_group_ancient_english }
            multiply = same_language_group_choice_factor
        }
        else_if = {
            limit = { has_cultural_parameter = language_family_englic }
            multiply = same_language_family_choice_factor
        }
    }
    color = { 226 164 153 }
}

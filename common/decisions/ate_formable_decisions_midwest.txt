# Form Ohio
form_ohio_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}
	decision_group_type = major
	ai_check_interval = 8
	desc = form_ohio_decision_desc

	is_shown = {
		is_ruler = yes
		is_landed = yes
		OR = {
			is_independent_ruler = yes
			OR = { 
				has_title = title:e_usa
				has_title = title:e_steel_belt
				has_title = title:e_great_lakes
			}
			liege ?= { has_title = title:e_usa }
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:form_ohio_decision
			}
		}
		NOT = { exists = title:k_ohio.holder }
		OR = {
			# Must be in the vague region.
			AND = {
				OR = { 
					has_title = title:e_usa
					has_title = title:e_steel_belt
					has_title = title:e_great_lakes
				}
				any_vassal_or_below = {
					any_held_title = {
						tier = tier_county
						title_province = { geographical_region = custom_ohio }
					}
				}
			}
			any_held_title = {
				tier = tier_county
				title_province = { geographical_region = custom_ohio }
			}
		}	
	}

	is_valid = {

		trigger_if = {
			limit = {
				is_independent_ruler = no
				liege ?= { has_title = title:e_usa }
			}
			custom_tooltip = {
				text = ate_vassal_ask_president_form
				always = no
			}
		}

		OR = {
			highest_held_title_tier <= tier_kingdom
			OR = { 
				has_title = title:e_usa
				has_title = title:e_steel_belt
				has_title = title:e_great_lakes
			}
		}

		trigger_if = {
			limit = { is_independent_ruler = no }
			custom_tooltip = {
				text = form_ohio_decision_tt
				any_county_in_region = {
					region = custom_ohio
					percent >= ate_formable_target_percent_decimal_value
					OR = {
						holder = root
						holder.liege = root
						holder.liege.liege = root
					}
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = form_ohio_decision_tt
				any_county_in_region = {
					region = custom_ohio
					percent >= ate_formable_target_percent_decimal_value
					holder.top_liege = root
				}
			}
		}
	}

	effect = {
		hidden_effect = {
			form_ohio_decision_effects = yes
		}
		if = {
			limit = { NOT = { has_title = title:e_usa } }
			give_nickname = nick_the_restorer
		} 

		#Events
		
		trigger_event = ate_form.0022
	}

	cost = { 
		gold = { value = 300 } 
		prestige = { value = 1000 }
	}

	ai_potential = {
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
}
